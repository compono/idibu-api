<h1>
	Example</h1>
<h2>
	Request</h2>

```bash
POST https://ws.idibu.com/ws/rest/v1/reports/(REPORT_ID)?hash=
```

<h2>BASIC:</h2>

```xml

<idibu>
    <date-range>
        <from>2015-05-01</from>
        <to>2015-05-28</to>
    </date-range>
    <profiles>
        <offices />
        <teams />
        <profiles />
	<include-inactive>no</include-inactive>
    </profiles>
    <boards />
    <email>yourclient@hisserver.com</email>
</idibu>

```

<h2>WITH MORE PARAMETERS:</h2>

```xml

<idibu>
    <date-range>
        <from>2015-05-01</from>
        <to>2015-05-28</to>
    </date-range>
    <profiles>
        <offices>
            <office>1000</office>
        </offices>
        <teams>
            <team>1001</team>
            <team>1002</team>
            <team>1003</team>
        </teams>
        <profiles>
            <profile>1004</profile>
            <profile>1005</profile>
            <profile>1006</profile>
        </profiles>
        <include-inactive>no</include-inactive>
    </profiles>
    <boards>
        <board>1599</board>
    </boards>
    <email>yourclient@hisserver.com</email>
    <watchdog>
        <name>new_watchdog</name>
        <frequency>monthly</frequency>
        <range>month</range>
    </watchdog>
</idibu>

```

<h2>
	Response</h2>
	
	
```xml

<?xml version="1.0" encoding="UTF-8"?>
<idibu generator="idibu" version="1.0">
  <response>
    <message>Report queued for processing</message>
  </response>
  <status>success</status>
</idibu>

```
